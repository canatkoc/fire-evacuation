160 void drawCorridors(const FireSimulator& sim, const Layout& layout) {
161 const Corridor* c = sim.corridorsHead();
162 while (c != nullptr) {
163 // Çift yönlü topoloji: aynı kenarı iki kez çizmemek için
164 // yalnız adress sıralaması tutarsa atlamak yerine yarı saydam
165 // çizip üst üste bindirmeye izin veriyoruz. Performansa zararsız.
166 int x1, y1, x2, y2;
167 if (layout.get(c->from->id, x1, y1) &&
168 layout.get(c->to->id, x2, y2)) {
169 if (c->blocked) {
170 SDL_SetRenderDrawColor(sdl_renderer_, 90, 30, 30, 255);
171 } else {
172 const Uint8 r = (Uint8)(120 + 120 * c->fire_intensity);
173 const Uint8 g = (Uint8)(120 - 80 * c->fire_intensity);
174 const Uint8 b = (Uint8)(120 - 80 * c->fire_intensity);
175 SDL_SetRenderDrawColor(sdl_renderer_, r, g, b, 255);
176 }
177 // Kalın çizgi: paralel iki-üç çizgi
178 for (int off = -1; off <= 1; ++off) {
179 SDL_RenderDrawLine(sdl_renderer_,
180 x1, y1 + off, x2, y2 + off);
181 SDL_RenderDrawLine(sdl_renderer_,
182 x1 + off, y1, x2 + off, y2);
183 }
184 }
185 c = c->next_global;

205 void fillNode(const SDL_Rect& r, const CellNode* n) {
206 Uint8 base_r = 70, base_g = 110, base_b = 180;
207 switch (n->type) {
208 case Graph::NodeType::Room: base_r=70; base_g=110; base_b=180; break;
209 case Graph::NodeType::Hallway: base_r=120; base_g=120; base_b=130; break;
210 case Graph::NodeType::Stairwell: base_r=210; base_g=180; base_b=80; break;
211 case Graph::NodeType::Exit: base_r=60; base_g=180; base_b=90; break;
212 default: base_r=100; base_g=100; base_b=100; break;
213 }
214 // Yangın overlay (kırmızı bias)
215 double t = n->fire_intensity;
216 Uint8 rr = (Uint8)(base_r * (1.0 - t) + 230 * t);
217 Uint8 gg = (Uint8)(base_g * (1.0 - t) + 60 * t);
218 Uint8 bb = (Uint8)(base_b * (1.0 - t) + 30 * t);
219 SDL_SetRenderDrawColor(sdl_renderer_, rr, gg, bb, 255);
220 SDL_RenderFillRect(sdl_renderer_, &r);
221 }

242 bool computePersonScreenPos(const Person* p, const Layout& layout,
243 int& out_x, int& out_y) {
244 int sx, sy;
245 if (!layout.get(p->current_location, sx, sy)) return false;
246
247 if (p->on_edge) {
248 int tx, ty;
249 if (!layout.get(p->next_location, tx, ty)) {
250 out_x = sx; out_y = sy; return true;
251 }
252 double frac = (p->edge_traversal_time > 0)
253 ? static_cast<double>(p->progress_on_edge) /
254 static_cast<double>(p->edge_traversal_time)
255 : 0.0;
256 if (frac < 0.0) frac = 0.0;
257 if (frac > 1.0) frac = 1.0;
258 out_x = static_cast<int>(sx + (tx - sx) * frac);
259 out_y = static_cast<int>(sy + (ty - sy) * frac);
260 } else {
261 out_x = sx;
262 out_y = sy;
263 }
264 return true;
265 }

267 void fillCircle(int cx, int cy, int radius) {
268 for (int dy = -radius; dy <= radius; ++dy) {
269 for (int dx = -radius; dx <= radius; ++dx) {
270 if (dx * dx + dy * dy <= radius * radius) {
271 SDL_RenderDrawPoint(sdl_renderer_, cx + dx, cy + dy);
272 }
