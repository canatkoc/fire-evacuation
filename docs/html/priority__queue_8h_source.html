131 void bubbleDown(HeapNode* node) {
132 while (node != nullptr) {
133 HeapNode* best = node;
134 if (node->left != nullptr && compare_(node->left->value, best->value)) best = node->left;
135 if (node->right != nullptr && compare_(node->right->value, best->value)) best = node->right;
136 if (best == node) return;
137 T temp = node->value;
138 node->value = best->value;
139 best->value = temp;
140 node = best;

323template <typename T, typename Compare>
324typename PriorityQueue<T, Compare>::HeapNode*
325PriorityQueue<T, Compare>::findLastNodeLevelOrder(HeapNode* node) const {
326 if (node == nullptr) return nullptr;
327
328 HeapNode* bfs_head = node;
329 HeapNode* bfs_tail = node;
330 node->next_insert = nullptr;
331 HeapNode* last = nullptr;
332
333 while (bfs_head != nullptr) {
334 HeapNode* current = bfs_head;
335 bfs_head = bfs_head->next_insert;
336 if (bfs_head == nullptr) bfs_tail = nullptr;
337 current->next_insert = nullptr;
338 last = current;
339
340 auto enq = [&](HeapNode* n) {
341 n->next_insert = nullptr;
342 if (bfs_tail == nullptr) { bfs_head = bfs_tail = n; }
343 else { bfs_tail->next_insert = n; bfs_tail = n; }
344 };
345 if (current->left != nullptr) enq(current->left);
346 if (current->right != nullptr) enq(current->right);
347 }
348 return last;

351template <typename T, typename Compare>
352void PriorityQueue<T, Compare>::pop() {
353 if (root_ == nullptr) throw std::out_of_range("PriorityQueue is empty.");
354
355 if (size_ == 1) {
356 delete root_;
357 root_ = nullptr;
358 insertion_queue_head_ = nullptr;
359 insertion_queue_tail_ = nullptr;
360 size_ = 0;
361 return;
362 }
363
364 HeapNode* last = findLastNodeLevelOrder(root_);
365 root_->value = last->value;
366
367 HeapNode* parent = last->parent;
368 if (parent != nullptr) {
369 if (parent->right == last) parent->right = nullptr;
370 else parent->left = nullptr;
371 }
372 delete last;
373 --size_;
374
375 // Rebuild the insertion-order queue via BFS over the remaining tree.
376 insertion_queue_head_ = nullptr;
377 insertion_queue_tail_ = nullptr;
378 if (root_ != nullptr) {
379 HeapNode* bfs_head = root_;
380 HeapNode* bfs_tail = root_;
381 root_->next_insert = nullptr;
382
383 while (bfs_head != nullptr) {
384 HeapNode* current = bfs_head;
385 bfs_head = bfs_head->next_insert;
386 if (bfs_head == nullptr) bfs_tail = nullptr;
387 current->next_insert = nullptr;
388
389 if (current->left == nullptr || current->right == nullptr) {
390 enqueueInsertionNode(current);
391 }
392
393 auto enq = [&](HeapNode* n) {
394 n->next_insert = nullptr;
395 if (bfs_tail == nullptr) { bfs_head = bfs_tail = n; }
396 else { bfs_tail->next_insert = n; bfs_tail = n; }
397 };
398 if (current->left != nullptr) enq(current->left);
399 if (current->right != nullptr) enq(current->right);
400 }

403 bubbleDown(root_);
