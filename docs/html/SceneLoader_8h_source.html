102

105 if (n->hasCoordinates()) {

107 continue;
108 }
109 int row = index / cols;
110 int col = index % cols;
111 int rows = (count + cols - 1) / cols;
112 int gx = margin_x +
113 (cols <= 1 ? span_x / 2 : col * span_x / (cols - 1));
114 int gy = margin_y +
