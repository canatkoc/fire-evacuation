68
69 // Her frame'de geçen süre (saniye) ile çağır. State==Running iken
70 // dahili biriktirici tick_interval_'i aştıkça sim->step() tetiklenir.
71 void update(double dt_seconds) {
72 if (state_ != State::Running) return;
73 if (sim_->simulationFinished()) {
74 state_ = State::Finished;
75 return;
76 }
77
78 accumulator_ += dt_seconds;
79 // Bir frame'de birden fazla tick birikebilir (örn. UI takılırsa).
80 while (accumulator_ >= tick_interval_ && state_ == State::Running) {
81 sim_->step();
82 accumulator_ -= tick_interval_;
83 if (sim_->simulationFinished()) {
84 state_ = State::Finished;
85 break;
86 }
87 }
88 }
