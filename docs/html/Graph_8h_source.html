384 bool addEdge(const std::string& from_id,
385 const std::string& to_id,
386 int traversal_time,
387 int capacity,
388 bool bidirectional = true) {
389 if (traversal_time < 0 || capacity < 0) {
390 throw std::invalid_argument("Traversal time and capacity must be non-negative.");
391 }
392
393 Vertex* from = findVertex(from_id);
394 Vertex* to = findVertex(to_id);
395 if (from == nullptr || to == nullptr) {
396 return false;
397 }
398
399 if (findEdge(from, to) == nullptr) {
400 Edge* edge = new Edge(to, traversal_time, capacity);
401 edge->next = from->adjacency;
402 from->adjacency = edge;
403 }
404
405 if (bidirectional && findEdge(to, from) == nullptr) {
406 Edge* reverse = new Edge(from, traversal_time, capacity);
407 reverse->next = to->adjacency;
408 to->adjacency = reverse;
409 }
410
411 return true;
412 }

471 PathResult shortestPath(const std::string& start_id, const std::string& target_id) const {
472 PathResult result;
473 Vertex* start = findVertex(start_id);
474 Vertex* target = findVertex(target_id);
475
476 if (start == nullptr || target == nullptr || start->blocked || target->blocked) {
477 return result;
478 }
479
480 DistRecord* table = buildDistanceTable();
481 DistRecord* start_rec = recordFor(table, start);
482 if (start_rec == nullptr) {
483 freeDistanceTable(table);
484 return result;
485 }
486
487 start_rec->distance = 0;
488
489 PriorityQueue<QueueItem, QueueCompare> pq;
490 pq.push({start, 0});
491
492 while (!pq.empty()) {
493 QueueItem current_item = pq.top();
494 pq.pop();
495
496 DistRecord* current_rec = recordFor(table, current_item.vertex);
497 if (current_rec == nullptr || current_rec->visited) {
498 continue;
499 }
500 if (current_item.distance > current_rec->distance) {
501 continue;
502 }
503
504 current_rec->visited = true;
505 if (current_item.vertex == target) {
506 break;
507 }
508
509 Edge* edge = current_item.vertex->adjacency;
510 while (edge != nullptr) {
511 if (!edge->blocked && !edge->to->blocked) {
512 DistRecord* next_rec = recordFor(table, edge->to);
513 if (next_rec != nullptr && !next_rec->visited &&
514 current_rec->distance != std::numeric_limits<int>::max()) {
515 const int candidate = current_rec->distance + edge->traversal_time;
516 if (candidate < next_rec->distance) {
517 next_rec->distance = candidate;
518 next_rec->previous = current_item.vertex;
519 pq.push({edge->to, candidate});
520 }
521 }
522 }
523 edge = edge->next;
524 }
525 }
526
527 DistRecord* target_rec = recordFor(table, target);
528 if (target_rec != nullptr && target_rec->distance != std::numeric_limits<int>::max()) {
529 result.reachable = true;
530 result.total_cost = target_rec->distance;
531
532 Vertex* walk = target;
533 while (walk != nullptr) {
534 result.path.push_front(walk->id);
535 DistRecord* walk_rec = recordFor(table, walk);
536 walk = (walk_rec == nullptr) ? nullptr : walk_rec->previous;
537 }
538 }
539
540 freeDistanceTable(table);
541 return result;
542 }

551 PathResult shortestPathToNearestExit(const std::string& start_id) const {
552 PathResult best;
553 Vertex* start = findVertex(start_id);
554 if (start == nullptr || start->blocked) {
555 return best;
556 }
557
558 Vertex* current = vertices_head_;
559 while (current != nullptr) {
560 if (current->type == NodeType::Exit && !current->blocked) {
561 PathResult candidate = shortestPath(start_id, current->id);
562 if (candidate.reachable &&
563 (!best.reachable || candidate.total_cost < best.total_cost)) {
564 best = candidate;
565 }
566 }
567 current = current->next;
568 }
569 return best;
570 }
