211template<typename T>
212void LinkedList<T>::remove(const T& value) {
213 if (is_empty()) return;
214 Node<T>* current = head;
215 Node<T>* previous = nullptr;
216 while (current != nullptr) {
217 if (current->data == value) {
218 if (previous == nullptr) head = current->next;
219 else previous->next = current->next;
220 if (current == tail) tail = previous;
221 delete current;
222 --size;
223 return;
224 }
225 previous = current;
226 current = current->next;
227 }
